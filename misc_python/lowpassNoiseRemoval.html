import numpy as np
import soundfile as sf
from scipy.signal import freqz
import math
import matplotlib.pyplot as plt

data, sr = sf.read('P_9_2.wav')

# Setting up the lowpass filter
fc = 7500
L = 101
M = L -1
ft = fc / sr

l_filt = np.array([])

#building the low pass filter from equation
for n in range(L):
    if n == M/2:
        x = 2 * ft
        l_filt = np.append(l_filt, x)
    else:
        x = ((math.sin(2 * np.pi * ft * (n - M / 2))) / (np.pi * (n - M / 2)))
        l_filt = np.append(l_filt, x)

x1, y1 = freqz(l_filt)

hamming = np.array([])

#building the hamming window
for n in range(L):
    x = (0.54 - 0.46 * (math.cos((2 * np.pi * n) / M)))
    hamming = np.append(hamming, x)

windowed = np.multiply(hamming, l_filt)

print(l_filt)
print(hamming)

x2, y2 = freqz(windowed)

#plotting the coefficients
plt.figure(1)
plt.subplot(211)
plt.title("Lowpass Filter Frequency Response")
plt.plot(x1, abs(y1))

plt.subplot(212)
plt.title("Windowed Filter Frequency Response")
plt.plot(x2, abs(y2))

#applying the windowed filter
filtered = np.convolve(windowed, data)
sf.write('cleanMusic.wav', filtered, sr)

plt.show()
