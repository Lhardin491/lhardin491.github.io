import numpy as np
import soundfile as sf
from scipy.signal import freqz
import math
import matplotlib.pyplot as plt

data, sr = sf.read('P_9_2.wav')

data = np.fft.fft(data)

#plot the magnitudes
plt.figure(1)
plt.subplot(211)
plt.plot(data)
plt.title("FFT of Original Signal")
plt.ylim(0)

#find midpoint and offset and set values to zero
midpoint = len(data) / 2
offset = 10000

for n in range(int(midpoint - offset), int(midpoint + offset)):
    data[n] = 0

#plot new magnitudes
plt.subplot(212)
plt.plot(data.real)
plt.title("FFT of Modified Signal")
plt.ylim(0)

#transform and write cleaned file
data = np.fft.ifft(data)
sf.write('cleanMusic.wav', data.real, sr)

plt.show()
