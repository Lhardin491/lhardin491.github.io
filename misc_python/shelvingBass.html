import numpy as np
import soundfile as sf
import matplotlib.pyplot as plt
import math

with open('shelvingConfig.txt') as f:
    config = f.readlines()

config = [x.strip() for x in config]

data, samplerate = sf.read(config[0])
sr = np.arange(samplerate)
g = int(config[1])
fc = int(config[2])
N = len(data)
theta = 2 * np.pi * fc / samplerate
mew = math.pow(10, (g / 20))

gamma = (1 - (4 / (1 + mew)) * math.tan(theta / 2))
alpha = (1 - gamma) / 2

#build the filter
u = np.array([])
y = np.array([])

for n in range(N):
    if n == 0:
        x = alpha * (data[n])
        u = np.append(u, x)
    else:
        x = alpha * (data[n] + data[n-1]) + (gamma * (u[n-1]))
        u = np.append(u, x)

for n in range(N):
    x = data[n] + (mew - 1) * u[n]
    y = np.append(y, x)

sf.write('shelvingOutput.wav', y, samplerate)

N = len(np.fft.fft(data))

#Plot the signal
plt.figure(1)
plt.subplot(121)
plt.title("Original Signal")
plt.plot(sr[0:int(N/4)], np.fft.fft(data)[0:int(N/4)])
plt.ylim(0, np.amax(np.fft.fft(data)) + 100)
plt.xlim(0, int(N/4))

plt.subplot(122)
plt.title("Modified Signal")
plt.plot(sr[0:int(N/4)], np.fft.fft(y)[0:int(N/4)])
plt.ylim(0, np.amax(np.fft.fft(data)) + 100)
plt.xlim(0, int(N/4))

plt.show()
